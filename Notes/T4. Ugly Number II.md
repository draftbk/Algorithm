# T4. Ugly Number II

> Medium 

用三个指针和一个 list 保存，然后 2，3，5 都乘指针对应的值，然后取最小的那个加入 list, 并把它的指针+1。

```
def nthUglyNumber(self, n):
    # write your code here
    if n<=1:
        return n
    mylist=[1]
    i2=0
    i3=0
    i5=0
    while len(mylist)<n:
        v2=mylist[i2]*2
        v3=mylist[i3]*3
        v5=mylist[i5]*5
        choose=min(v2,v3,v5)
        print(choose)
        if choose==v2:
            i2=i2+1
            if choose not in mylist:
                mylist.append(choose)
        elif choose==v3:
            i3=i3+1
            if choose not in mylist:
                mylist.append(choose)
        elif choose==v5:
            i5=i5+1
            if choose not in mylist:
                mylist.append(choose)
    return mylist.pop()
```
